    // debug build

    // release build
    cmake -G "MinGW Makefiles" -D CMAKE_BUILD_TYPE=Release -DCMAKE_INCLUDE_PATH=\PATH\TO\_deps-win32\mingw\include -DCMAKE_LIBRARY_PATH=\PATH\TO\_deps-win32\mingw\lib\release -DCMAKE_FRAMEWORK_PATH=\PATH\TO\_deps-win32\mingw\share ..\
    mingw32-make install -j9
