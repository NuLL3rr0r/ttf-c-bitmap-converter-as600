1. In MSYS shell:

    mkdir debug release

    # debug build
    cd debug && ../configure --with-our-malloc16 --with-windows-f77-mangling --enable-shared --disable-static --enable-threads --with-combined-threads --enable-portable-binary --enable-sse2 --with-incoming-stack-boundary=2 --prefix=/d/build/debug CPPFLAGS=-DDEBUG CFLAGS="-g -O0" CXXFLAGS="-g -O0"
    make -j9 && make install

    # release build
    cd release && ../configure --with-our-malloc16 --with-windows-f77-mangling --enable-shared --disable-static --enable-threads --with-combined-threads --enable-portable-binary --enable-sse2 --with-incoming-stack-boundary=2 --prefix=/d/build/release CPPFLAGS=-DNDEBUG
    make -j9 && make install

2. Copy the following:
    D:\build\release\include\* to our include dir
    D:\build\release\bin\*.dll to our rel-lib dir
    D:\build\debug\bin\*.dll to our deb-lib dir
