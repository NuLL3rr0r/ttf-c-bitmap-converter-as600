1. In MSYS shell:

    mkdir _debug _release

    # _debug build
    cd _debug && ../configure --with-our-malloc16 --with-windows-f77-mangling --enable-shared --disable-static --enable-threads --with-combined-threads --enable-portable-binary --enable-sse2 --with-incoming-stack-boundary=2 --prefix=/d/build/_debug-fftw3 CPPFLAGS=-DDEBUG CFLAGS="-g -O0" CXXFLAGS="-g -O0"
    make -j9 && make install

    # release build
    cd _release && ../configure --with-our-malloc16 --with-windows-f77-mangling --enable-shared --disable-static --enable-threads --with-combined-threads --enable-portable-binary --enable-sse2 --with-incoming-stack-boundary=2 --prefix=/d/build/_release-fftw3 CPPFLAGS=-DNDEBUG
    make -j9 && make install

2. Copy the following:
    D:\build\release\include\* to our include dir
    D:\build\release\bin\*.dll to our rel-lib dir
    D:\build\debug\bin\*.dll to our deb-lib dir
